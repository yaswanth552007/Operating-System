#include <stdio.h>

int main() {
    int f[50], p, i, st, len, j;
    for(i=0;i<50;i++) f[i]=0;
    printf("Enter starting block and length of file: ");
    scanf("%d%d", &st, &len);
    for(j=st; j<(st+len); j++) {
        if(f[j]==0) {
            f[j]=1;
            printf("Block %d allocated.\n", j);
        } else {
            printf("Block %d already allocated.\n", j);
            break;
        }
    }
    return 0;
}
