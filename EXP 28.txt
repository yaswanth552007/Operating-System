#include <stdio.h>
#include <string.h>

int main(int argc, char *argv[]){
    if(argc < 3){ printf("Usage: %s pattern filename\n", argv[0]); return 1; }
    char *pattern = argv[1];
    FILE *f = fopen(argv[2], "r");
    if(!f){ perror("fopen"); return 1; }
    char line[512]; int lineno=1;
    while(fgets(line, sizeof(line), f)){
        if(strstr(line, pattern)) printf("%s:%d: %s", argv[2], lineno, line);
        lineno++;
    }
    fclose(f);
    return 0;
}
