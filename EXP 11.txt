#include <stdio.h>
#include <pthread.h>
void *thread_func(void *arg) {
    printf("Hello from thread %d\n", *(int *)arg);
    return NULL;
}
int main() {
    pthread_t t1, t2;
    int a = 1, b = 2;
    pthread_create(&t1, NULL, thread_func, &a);
    pthread_create(&t2, NULL, thread_func, &b);
    pthread_join(t1, NULL);
    pthread_join(t2, NULL);
    return 0;
}
